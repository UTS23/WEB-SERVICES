@startuml
title Diagrama de clases
class Inmueble {
  + id: Long
  + referencia: String
  + tipo: String
  + direccion: String
  + precio: Double
  + disponible: Boolean
}

class Oficina {
  + id: Long
  + ubicacion: String
  + nombre: String
  + descripcion: String
}

class ZonaCiudad {
  + id: Long
  + nombre: String
  + descripcion: String
  + inmuebles: List<Inmueble>
}

class Cliente {
  + id: Long
  + identificacion: String
  + nombre: String
  + visitas: List<Visita>
  + inmueblesFavoritos: List<Inmueble>
}

class Visita {
  + id: Long
  + fecha: Date
  + cliente: Cliente
  + inmueble: Inmueble
}

Cliente --|> Inmueble: inmueblesFavoritos
Cliente "1" -- "many" Visita : visitas
Visita "1" -- "1" Inmueble : inmueble
ZonaCiudad "1" -- "many" Inmueble : inmuebles
Oficina "1" -- "many" Inmueble : ubicacion
@enduml
